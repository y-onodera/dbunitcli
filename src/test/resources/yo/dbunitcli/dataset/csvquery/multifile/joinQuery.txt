SELECT
    table1.*
FROM
    CSVREAD('target/test-classes/yo/dbunitcli/dataset/csvquery/csv/header.csv',null,'charset=UTF-8') table1
    inner join
    CSVREAD('target/test-classes/yo/dbunitcli/dataset/csvquery/csv/detail.csv',null,'charset=UTF-8') table2
    on table1.key = table2.key
WHERE
    table2.detailno=2
ORDER BY
    table1.key desc
;
